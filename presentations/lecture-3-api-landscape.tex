% -------------------------------------------------------
% Slide 3: UK Parliament API
% -------------------------------------------------------
\begin{frame}{UK Parliament API \hfill \faIcon{landmark}}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \begin{description}[\textbf{Use case}]
                \item[\textbf{What}] Bills, debates, MPs, divisions --- the full record of parliamentary activity
                \item[\textbf{Auth}] None
                \item[\textbf{Format}] JSON
                \item[\textbf{Use case}] Political science, legislative studies, democratic representation
            \end{description}
        \end{column}
        \begin{column}{0.42\textwidth}
            \begin{exampleblock}{Example endpoint}
                {\footnotesize\ttfamily
                members-api.parliament.uk/\\api/Members/Search?\\Name=Smith}
            \end{exampleblock}
            \vspace{0.2cm}
            \begin{block}{Documentation}
                {\footnotesize \faIcon{book} \url{https://members-api.parliament.uk/index.html}}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

% -------------------------------------------------------
% Slide 4: World Bank Indicators API
% -------------------------------------------------------
\begin{frame}{World Bank Indicators API \hfill \faIcon{globe}}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \begin{description}[\textbf{Use case}]
                \item[\textbf{What}] Development indicators (GDP, poverty, education, health) for 200+ countries and territories
                \item[\textbf{Auth}] None
                \item[\textbf{Format}] JSON
                \item[\textbf{Use case}] Comparative politics, development studies, economics
            \end{description}
        \end{column}
        \begin{column}{0.42\textwidth}
            \begin{exampleblock}{Example endpoint}
                {\footnotesize\ttfamily
                api.worldbank.org/v2/\\country/GBR/indicator/\\NY.GDP.PCAP.CD?\\format=json}
            \end{exampleblock}
            \vspace{0.2cm}
            \begin{block}{Documentation}
                {\footnotesize \faIcon{book} \url{https://datahelpdesk.worldbank.org/knowledgebase/articles/889392}}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{OpenAI / Anthropic API \hfill \faIcon{robot}}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \begin{description}[\textbf{Use case}]
                \item[\textbf{What}] LLM-as-a-service: text generation, classification, embeddings, summarisation
                \item[\textbf{Auth}] API key (paid)
                \item[\textbf{Format}] JSON
                \item[\textbf{Use case}] Text annotation, data classification, synthetic data generation
            \end{description}
        \end{column}
        \begin{column}{0.42\textwidth}
            \begin{exampleblock}{Example endpoint}
                {\footnotesize\ttfamily
                api.openai.com/v1/\\chat/completions}

% -------------------------------------------------------
% Slide 7: OpenStreetMap / Nominatim API
% -------------------------------------------------------
\begin{frame}{OpenStreetMap / Nominatim API \hfill \faIcon{map-marked-alt}}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \begin{description}[\textbf{Use case}]
                \item[\textbf{What}] Geospatial data, geocoding (address \(\to\) coordinates), reverse geocoding (coordinates \(\to\) address)
                \item[\textbf{Auth}] None (rate limited: 1 req/sec)
                \item[\textbf{Format}] JSON
                \item[\textbf{Use case}] Spatial analysis, urban studies, geography
            \end{description}
        \end{column}
        \begin{column}{0.42\textwidth}
            \begin{exampleblock}{Example endpoint}
                {\footnotesize\ttfamily
                nominatim.openstreetmap.org/\\search?q=Edinburgh\&\\format=json}
            \end{exampleblock}
            \vspace{0.2cm}
            \begin{block}{Documentation}
                {\footnotesize \faIcon{book} \url{https://nominatim.org/release-docs/latest/api/Overview/}}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Finding More APIs}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \begin{block}{Curated directory}
                {\large \faIcon{github}} \textbf{public-apis} on GitHub\\[0.2cm]
                {\small A collective list of free APIs for development, research, and fun.}\\[0.3cm]
                {\footnotesize \url{https://github.com/public-apis/public-apis}}
            \end{block}
            \vspace{0.3cm}
            \begin{block}{Other resources}
                \begin{itemize}
                    \small
                    \item \textbf{RapidAPI Hub} --- search and test APIs
                    \item \textbf{ProgrammableWeb} --- API directory
                    \item \textbf{Google Dataset Search} --- find open datasets
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.42\textwidth}
            \begin{alertblock}{Before committing to an API}
                \begin{enumerate}
                    \small
                    \item \faIcon{book} \textbf{Documentation quality} --- is it clear and up to date?
                    \item \faIcon{clock} \textbf{Rate limits} --- can you collect data at a reasonable pace?
                    \item \faIcon{file-contract} \textbf{Terms of Service} --- does the ToS allow research use?
                    \item \faIcon{lock} \textbf{Authentication} --- is access free or paid?
                    \item \faIcon{sync-alt} \textbf{Stability} --- is the API actively maintained?
                \end{enumerate}
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}
